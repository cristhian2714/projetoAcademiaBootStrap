Importante: nunca salve senhas em texto puro no banco. Sempre use password_hash() para salvar e password_verify() para comparar.

<?php
namespace Projeto\DAO; // "endereço" da classe — evita conflito de nomes

class Conexao {
    function conectar() {
        try {
            // mysqli_connect(host, usuario, senha, banco)
            $conn = mysqli_connect('127.0.0.1', 'root', '', 'academiaDefault');
            if ($conn) {
                mysqli_set_charset($conn, "utf8mb4"); // suporte a acentos
                return $conn; // retorna a conexão para quem chamou
            }
        } catch (Exception $erro) {
            echo "Algo deu errado na conexão <br>$erro";
        }
        return null;
    }//fim conectar
}//fim classe
?>

6. DAO/Consultar.php — Consultas SELECT

function consultarAlunos(Conexao $conexao) {
    $conn   = $conexao->conectar();
    // LEFT JOIN une duas tabelas — traz o nome do plano junto com o aluno
    $sql    = "SELECT a.*, p.nome as plano_nome
               FROM aluno a
               LEFT JOIN plano p ON a.codigoPlano = p.codigo
               ORDER BY a.codigo DESC";
    $result = mysqli_query($conn, $sql);   // executa a query
    $dados  = [];
    while ($row = mysqli_fetch_assoc($result)) { // lê linha a linha
        $dados[] = $row; // adiciona ao array de resultados
    }//fim while
    return $dados;
}//fim consultarAlunos

// Consulta aluno por e-mail (usada no login)
function consultarAlunoPorEmail(Conexao $conexao, string $email) {
    $conn  = $conexao->conectar();
    $email = mysqli_real_escape_string($conn, $email); // sanitiza contra SQL Injection
    $sql   = "SELECT a.*, p.nome as plano_nome
              FROM aluno a LEFT JOIN plano p ON a.codigoPlano = p.codigo
              WHERE a.email = '$email'";
    $result = mysqli_query($conn, $sql);
    return mysqli_fetch_assoc($result); // retorna só 1 registro (ou false)
}//fim consultarAlunoPorEmail

7. DAO/Inserir.php — Inserção INSERT

function inserirAluno(Conexao $conexao, $nome, $dtNascimento, $peso,
                      $altura, $objetivo, $email, $senha, $codigoPlano) {
    $conn      = $conexao->conectar();
    $nome      = mysqli_real_escape_string($conn, $nome);     // protege contra SQL Injection
    $objetivo  = mysqli_real_escape_string($conn, $objetivo);
    $email     = mysqli_real_escape_string($conn, $email);
    $senhaHash = password_hash($senha, PASSWORD_DEFAULT);      // nunca salva senha em texto puro!
    $sql       = "INSERT INTO aluno (nome, dtNascimento, peso, altura, objetivo, email, senha, codigoPlano)
                  VALUES ('$nome', '$dtNascimento', $peso, $altura, '$objetivo', '$email', '$senhaHash', $codigoPlano)";
    return mysqli_query($conn, $sql); // retorna true (sucesso) ou false (erro)
}//fim inserirAluno

8. DAO/Atualizar.php — Atualização UPDATE

function atualizarAluno(Conexao $conexao, $codigo, $nome, $dtNascimento,
                         $peso, $altura, $objetivo, $email, $codigoPlano, $senha = '') {
    $conn  = $conexao->conectar();
    $nome  = mysqli_real_escape_string($conn, $nome);
    $email = mysqli_real_escape_string($conn, $email);

    if (!empty($senha)) {
        // Atualiza COM nova senha
        $senhaHash = password_hash($senha, PASSWORD_DEFAULT);
        $sql = "UPDATE aluno SET nome='$nome', email='$email', senha='$senhaHash',
                codigoPlano=$codigoPlano WHERE codigo=$codigo";
    } else {
        // Mantém a senha atual (campo senha em branco na edição)
        $sql = "UPDATE aluno SET nome='$nome', email='$email',
                codigoPlano=$codigoPlano WHERE codigo=$codigo";
    }//fim if
    return mysqli_query($conn, $sql);
}//fim atualizarAluno
⚠️ O WHERE é obrigatório! Sem WHERE codigo = $codigo, o UPDATE alteraria todos os registros da tabela.
